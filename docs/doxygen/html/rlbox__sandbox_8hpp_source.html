53  // This variable tracks of the sandbox has already been created/destroyed.
54  // APIs in this class should be called only when the sandbox is created.
55  // However, it is expensive to check in APIs such as invoke or in the callback
56  // interceptor. What's more, there could be time of check time of use issues
57  // in the checks as well.
58  // In general, we leave it up to the user to ensure these APIs are never

79  inline constexpr void check_invoke_param_type_is_ok()
80  {
81  using T_NoRef = std::remove_reference_t<T>;
82 
83  if_constexpr_named(cond1, detail::rlbox_is_wrapper_v<T_NoRef>)
84  {
85  if_constexpr_named(
86  subcond1,
87  !std::is_same_v<T_Sbx, detail::rlbox_get_wrapper_sandbox_t<T_NoRef>>)
88  {
89  rlbox_detail_static_fail_because(
90  cond1 && subcond1,
91  "Mixing tainted data from a different sandbox types. This could "
92  "happen due to couple of different reasons.\n"
93  "1. You are using 2 sandbox types for example'rlbox_noop_sandbox' "
94  "and 'rlbox_lucet_sandbox', and are passing tainted data from one "
95  "sandbox as parameters into a function call to the other sandbox. "
96  "This is not allowed, unwrap the tainted data with copy_and_verify "
97  "or other unwrapping APIs first.\n"
98  "2. You have inadvertantly forgotten to set/remove "
99  "RLBOX_USE_STATIC_CALLS depending on the sandbox type. Some sandbox "
100  "types like rlbox_noop_sandbox require this to be set to a given "
101  "value, while other types like rlbox_lucet_sandbox, require this not "
102  "to be set.");
103  }
104  }
105  else if_constexpr_named(cond2,
106  std::is_null_pointer_v<T_NoRef> ||
107  detail::is_fundamental_or_enum_v<T_NoRef>)
108  {}
109  else

112  rlbox_detail_static_fail_because(

119  template<typename T>
120  inline auto invoke_process_param(T&& param)
121  {
122  check_invoke_param_type_is_ok<T>();
123 
124  using T_NoRef = std::remove_reference_t<T>;
125 
126  if constexpr (detail::rlbox_is_tainted_opaque_v<T_NoRef>) {
127  auto ret = from_opaque(param);
128  return ret.UNSAFE_sandboxed(*this);
129  } else if constexpr (detail::rlbox_is_wrapper_v<T_NoRef>) {
130  return param.UNSAFE_sandboxed(*this);
131  } else if constexpr (std::is_null_pointer_v<T_NoRef>) {
132  tainted<void*, T_Sbx> ret = nullptr;
133  return ret.UNSAFE_sandboxed(*this);
134  } else if constexpr (detail::is_fundamental_or_enum_v<T_NoRef>) {
135  // For unwrapped primitives, assign to a tainted var and then unwrap so
136  // that we adjust for machine model
137  tainted<T_NoRef, T_Sbx> ret = param;
138  return ret.UNSAFE_sandboxed(*this);
139  } else {
140  rlbox_detail_static_fail_because(detail::true_v<T_NoRef>, "Unknown case");
141  }

161  template<typename T_Ret, typename... T_Args>
162  static detail::convert_to_sandbox_equivalent_t<T_Ret, T_Sbx>
163  sandbox_callback_interceptor(
164  detail::convert_to_sandbox_equivalent_t<T_Args, T_Sbx>... args)
165  {
166  std::pair<T_Sbx*, void*> context =
167  T_Sbx::impl_get_executed_callback_sandbox_and_key();
168  auto& sandbox = *(reinterpret_cast<rlbox_sandbox<T_Sbx>*>(context.first));
169  auto key = context.second;
170 
171  using T_Func_Ret =
172  std::conditional_t<std::is_void_v<T_Ret>, void, tainted<T_Ret, T_Sbx>>;
173  using T_Func =
174  T_Func_Ret (*)(rlbox_sandbox<T_Sbx>&, tainted<T_Args, T_Sbx>...);
175  auto target_fn_ptr = reinterpret_cast<T_Func>(key);
176 
177  if constexpr (std::is_void_v<T_Func_Ret>) {
178  (*target_fn_ptr)(
179  sandbox,
180  sandbox.template sandbox_callback_intercept_convert_param<T_Args>(
181  sandbox, args)...);
182  return;
183  } else {
184  auto tainted_ret = (*target_fn_ptr)(
185  sandbox,
186  sandbox.template sandbox_callback_intercept_convert_param<T_Args>(
187  sandbox, args)...);
188 
189  using namespace detail;
190  convert_to_sandbox_equivalent_t<T_Ret, T_Sbx> ret;
191  convert_type<T_Sbx,
192  adjust_type_direction::TO_SANDBOX,
193  adjust_type_context::SANDBOX>(
194  ret,
195  tainted_ret.get_raw_value_ref(),
196  nullptr /* example_unsandboxed_ptr */,
197  &sandbox);
198  return ret;
